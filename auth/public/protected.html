async function refreshToken() { const refreshToken = localStorage.getItem('refreshToken'); if (!refreshToken) { window.location.href = '/login.html'; return; } try { const response = await fetch('/api/auth/refresh', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ refresh_token: refreshToken }) }); } catch (error) { console.error('Error refreshing token:', error); window.location.href = '/login.html'; } }
